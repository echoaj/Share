$now      = Get-Date
$start    = $now.AddMinutes(-1)

# trigger a file access so an event is created
Get-ChildItem -LiteralPath C:\Directory -Force | Out-Null
Start-Sleep -Seconds 2

$evt = Get-WinEvent -FilterHashtable @{ LogName='Security'; Id=4656; StartTime=$start } |
       Select-Object -First 1

$evt | Select-Object TimeCreated, Id, LogName, TaskDisplayName,
  @{ Name='Audit'; Expression={
        if ($_.KeywordsDisplayNames -contains 'Audit Failure') {'Audit Failure'}
        elseif ($_.KeywordsDisplayNames -contains 'Audit Success') {'Audit Success'}
        else { ($_.KeywordsDisplayNames -join ', ') }
     }},
  @{ Name='AccessReasons'; Expression={
        $xml = [xml]$_.ToXml()
        ($xml.Event.EventData.Data |
            Where-Object { $_.Name -in 'AccessReason','AccessReasons' }
        ).'#text' -join '; '
     }},
  @{ Name='ContainsDenied'; Expression={
        $xml = [xml]$_.ToXml()
        $reasons = ($xml.Event.EventData.Data |
            Where-Object { $_.Name -in 'AccessReason','AccessReasons' }
        ).'#text' -join '; '
        [bool]($reasons -match '(?i)\bdenied\b')  # case-insensitive
     }}
