
- name: Clear File Explorer history
  hosts: windows_hosts
  tasks:
    - name: Clear Recent Docs
      win_regedit:
        path: HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RecentDocs
        name: ""
        state: absent

    - name: Clear Run history in Start Menu
      win_regedit:
        path: HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RunMRU
        name: ""
        state: absent

    - name: Clear Address Bar history in File Explorer
      win_regedit:
        path: HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\TypedPaths
        name: ""
        state: absent

    - name: Notify Explorer to refresh
      win_command: tsdiscon
      ignore_errors: yes
