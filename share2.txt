 public UdpTest build() throws SocketException, IOException
        {
            NetworkInterface.getByName(incomingNetworkInterface).getInetAddresses().asIterator().next();
            UdpTest udpTest = new UdpTest();
            udpTest.name = name;
            udpTest.incoming = new DatagramSocket(0, getLocalAddressForNic(incomingNetworkInterface));
            udpTest.outgoingAddress = InetAddress.getByName(outgoingAddress);
            udpTest.incomingAddress = InetAddress.getByName(incomingAddress);
            udpTest.incomingPort = incomingPort;
            udpTest.outgoing = new MulticastSocket (outgoingPort);
            udpTest.outgoing.joinGroup(new InetSocketAddress(udpTest.outgoingAddress, 0), NetworkInterface.getByName(outgoingNetworkInterface));
            LOGGER.info("building test to send unicast to {}:{} receiving multicast on {}:{}", incomingAddress, incomingPort, outgoingAddress, outgoingPort);
            return udpTest;
        }
